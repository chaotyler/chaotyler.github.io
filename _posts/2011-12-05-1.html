json.stringify()方法用于将javascript的数据如对象或字符串转换成json对象使用：

var newUser = {};
newUser.type = "BUSINESS";
newUser.username = $('#uname').val();
newUser.password = $('#upassword').val();
newUser.email = $('#uemail').val();
newUser.category = "RESTAURANT";
newUser.name = $('#urealname').val();
newUser.zip = $('#uzip').val();
newUser.contractor = $('#ucontractor').val();
newUser.telephone = $('#utelephone').val();
newUser.mobile = $('#umobile').val();
newUser.fax = $('#ufax').val();

var userStr = JSON.stringify(newUser);

$.ajax({
         type: "PUT",
         url: "resources/user",
         contentType: "application/json",
         data: userStr,
         success: function(msg){
         alert("regist succeed" + msg);
}});
